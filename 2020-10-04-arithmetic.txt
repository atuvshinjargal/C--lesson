using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp3
{
    class Program
    {
        static void Main(string[] args)
        {
            double x=3, y=5;
            Console.Write("x+y=");
            Console.WriteLine(x + y);
            Console.Write("x-y=");
            Console.WriteLine(x - y);
            Console.Write("x*y=");
            Console.WriteLine(x * y);
            Console.Write("x/y=");
            Console.WriteLine(x/y);
            Console.Write("x%y=");
            Console.WriteLine(x % y);
            Console.Write("x++ =>");
            Console.WriteLine(++x);
            Console.Write("x--");
            Console.WriteLine(--x);
        }
    }
}
